# 🎒 学生端本地化功能对话实现文档

> **核心思路：对话即入口，本地即数据。**
>
> 学生端除了校园端提供的 MCP 工具（签到、课表、作业等），还有自己的**纯本地功能**——不依赖校园端，不经过 MCP 协议，数据存在本地 SQLite 中。Agent 聊天时也能操作这些本地功能。

---

## 🧱 整体设计

```
学生说 "帮我记个笔记"
        │
        ▼
┌────────────────────────────────┐
│         Agent 意图识别           │
│                                │
│  ├── 需要校园端数据？→ MCP 工具  │  ← 调校园端（上篇文档）
│  │                              │
│  └── 纯本地功能？→ 本地 API      │  ← 本次文档重点
│       ├── 📝 备忘录管理          │     直接调后端 REST，不走 MCP
│       ├── ⭐ 收藏课程            │
│       ├── 📋 操作记录查询         │
│       ├── 📅 课表缓存查询         │
│       └── 🔔 通知历史查询         │
└────────────────────────────────┘
        │
        ▼
  本地 SQLite 数据库
  (6 张表，零网络依赖)
```

## 🗄️ 本地数据库（6 张表）

| # | 表名 | 用途 | 是否暴露给 Agent |
|:--|:-----|:-----|:---------------|
| 1 | `memos` | 备忘录/学习笔记 | ✅ — "记个笔记" |
| 2 | `operation_logs` | 操作记录 | ✅ — "我的操作记录" |
| 3 | `favorite_courses` | 收藏课程 | ✅ — "收藏这门课" |
| 4 | `schedule_cache` | 课表离线缓存 | ✅ — "缓存课表" |
| 5 | `notification_history` | 通知本地缓存 | ✅ — "查看通知历史" |
| 6 | `notification_reads` | 通知已读标记 | ❌ — 系统内部使用 |

### 建表 SQL（最小集）

```sql
-- 1️⃣ 备忘录 — Agent 最常用的本地功能
CREATE TABLE memos (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    title       TEXT    DEFAULT '',
    content     TEXT    DEFAULT '',
    is_important INTEGER DEFAULT 0,  -- 0:普通 1:重要
    tags        TEXT    DEFAULT '',   -- 逗号分隔的标签，如 "学习,高数"
    created_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 2️⃣ 操作记录 — 记录所有操作（含 MCP 和本地）
CREATE TABLE operation_logs (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    action_type TEXT    NOT NULL,  -- checkin / memo / favorite / login / ...
    content     TEXT    NOT NULL,  -- 描述
    result      TEXT,              -- 操作结果
    source      TEXT    DEFAULT 'local',  -- 'local' 或 'mcp'
    created_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 3️⃣ 收藏课程
CREATE TABLE favorite_courses (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    course_name TEXT    NOT NULL,
    teacher     TEXT,
    course_id   INTEGER,           -- 校园端课程 ID（可为空，纯本地收藏不需要）
    added_at    DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 4️⃣ 课表缓存
CREATE TABLE schedule_cache (
    course_id    INTEGER PRIMARY KEY,
    name         TEXT NOT NULL,
    teacher      TEXT,
    location     TEXT,
    day_of_week  INTEGER,
    start_time   TEXT,
    end_time     TEXT,
    cached_at    DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 5️⃣ 通知历史
CREATE TABLE notification_history (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    campus_notif_id INTEGER,
    title           TEXT,
    content         TEXT,
    notif_type      TEXT,     -- grade_notice / leave_result / checkin_reminder / announcement
    is_read         INTEGER DEFAULT 0,
    received_at     DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 6️⃣ 通知已读标记（系统内部）
CREATE TABLE notification_reads (
    notification_id INTEGER PRIMARY KEY,
    read_at         DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 1️⃣ 备忘录路由 (`memos.py`)

```python
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from database.database import get_db
from database.models import Memo
from pydantic import BaseModel
from datetime import datetime

router = APIRouter(prefix="/api/local/memos", tags=["local-memos"])

class MemoCreate(BaseModel):
    title: str = ""
    content: str = ""
    is_important: bool = False
    tags: str = ""

class MemoUpdate(BaseModel):
    title: str | None = None
    content: str | None = None
    is_important: bool | None = None
    tags: str | None = None

@router.get("")
async def list_memos(tag: str = "", important_only: bool = False, db: Session = Depends(get_db)):
    """列出所有备忘（Agent 调 '我的备忘' 时用）"""
    query = db.query(Memo).order_by(Memo.updated_at.desc())
    if tag:
        query = query.filter(Memo.tags.contains(tag))
    if important_only:
        query = query.filter(Memo.is_important == 1)
    memos = query.limit(50).all()
    return {
        "data": [
            {
                "id": m.id,
                "title": m.title,
                "content": m.content[:100],  # 摘要
                "is_important": bool(m.is_important),
                "tags": m.tags.split(",") if m.tags else [],
                "time": m.updated_at.strftime("%m-%d %H:%M"),
            }
            for m in memos
        ],
        "total": len(memos),
    }

@router.post("")
async def create_memo(body: MemoCreate, db: Session = Depends(get_db)):
    """创建备忘（Agent 调 '记个笔记' 时用）"""
    memo = Memo(**body.model_dump())
    db.add(memo)
    db.commit()
    db.refresh(memo)

    # 写操作记录
    _log_action(db, "memo", f"创建备忘：{body.title or '无标题'}")

    return {"status": "ok", "id": memo.id}

@router.get("/{memo_id}")
async def get_memo(memo_id: int, db: Session = Depends(get_db)):
    """查看备忘详情"""
    memo = db.query(Memo).filter(Memo.id == memo_id).first()
    if not memo:
        return {"status": "error", "msg": "备忘不存在"}
    return {
        "data": {
            "id": memo.id,
            "title": memo.title,
            "content": memo.content,
            "is_important": bool(memo.is_important),
            "tags": memo.tags.split(",") if memo.tags else [],
            "created_at": memo.created_at.strftime("%m-%d %H:%M"),
            "updated_at": memo.updated_at.strftime("%m-%d %H:%M"),
        }
    }

@router.put("/{memo_id}")
async def update_memo(memo_id: int, body: MemoUpdate, db: Session = Depends(get_db)):
    """更新备忘（Agent 调 '修改备忘' 时用）"""
    memo = db.query(Memo).filter(Memo.id == memo_id).first()
    if not memo:
        return {"status": "error", "msg": "备忘不存在"}

    update_data = body.model_dump(exclude_none=True)
    for key, value in update_data.items():
        setattr(memo, key, value)
    memo.updated_at = datetime.now()
    db.commit()

    return {"status": "ok"}

@router.delete("/{memo_id}")
async def delete_memo(memo_id: int, db: Session = Depends(get_db)):
    """删除备忘（Agent 调 '删掉这个备忘' 时用）"""
    memo = db.query(Memo).filter(Memo.id == memo_id).first()
    if not memo:
        return {"status": "error", "msg": "备忘不存在"}
    db.delete(memo)
    db.commit()

    _log_action(db, "memo", f"删除备忘：{memo.title or '无标题'}")
    return {"status": "ok"}

def _log_action(db, action_type: str, content: str):
    """写操作记录"""
    from database.models import OperationLog
    db.add(OperationLog(action_type=action_type, content=content, result="ok"))
    db.commit()
```

### 2️⃣ 操作记录路由 (`logs.py`)

```python
@router.get("/api/local/logs")
async def list_logs(
    action_type: str = "",
    limit: int = 50,
    db: Session = Depends(get_db),
):
    """查询操作记录（Agent 调 '我的操作记录' 时用）"""
    query = db.query(OperationLog).order_by(OperationLog.created_at.desc())
    if action_type and action_type != "all":
        query = query.filter(OperationLog.action_type == action_type)
    logs = query.limit(limit).all()
    return {
        "data": [
            {
                "id": log.id,
                "action_type": log.action_type,
                "content": log.content,
                "result": log.result,
                "source": log.source,
                "time": log.created_at.strftime("%m-%d %H:%M"),
            }
            for log in logs
        ],
        "total": len(logs),
    }
```

### 3️⃣ 收藏课程路由 (`favorites.py`)

```python
@router.get("/api/local/favorites")
async def list_favorites(db: Session = Depends(get_db)):
    """列出收藏课程"""
    favs = db.query(FavoriteCourse).order_by(FavoriteCourse.added_at.desc()).all()
    return {"data": [{"id": f.id, "name": f.course_name, "teacher": f.teacher} for f in favs]}

@router.post("/api/local/favorites")
async def add_favorite(body: dict, db: Session = Depends(get_db)):
    """收藏课程"""
    existing = db.query(FavoriteCourse).filter(
        FavoriteCourse.course_name == body["course_name"]
    ).first()
    if existing:
        return {"status": "ok", "msg": "已收藏过"}
    fav = FavoriteCourse(**body)
    db.add(fav)
    db.commit()

    _log_action(db, "favorite", f"收藏课程：{body.get('course_name', '')}")
    return {"status": "ok", "id": fav.id}

@router.delete("/api/local/favorites/{fav_id}")
async def remove_favorite(fav_id: int, db: Session = Depends(get_db)):
    """取消收藏"""
    fav = db.query(FavoriteCourse).filter(FavoriteCourse.id == fav_id).first()
    if not fav:
        return {"status": "error", "msg": "未找到"}
    db.delete(fav)
    db.commit()
    return {"status": "ok"}
```

### Agent 后端的本地工具路由映射

```python
# backend/router/chat.py — 在 tool_call 循环中路由
TOOL_ROUTING = {
    # 本地功能 → 调本地 REST API
    "memos_list":        ("GET",  "/api/local/memos"),
    "memos_create":      ("POST", "/api/local/memos"),
    "memos_delete":      ("DELETE","/api/local/memos/{memo_id}"),
    "logs_query":        ("GET",  "/api/local/logs"),
    "favorites_list":    ("GET",  "/api/local/favorites"),
    "favorites_add":     ("POST", "/api/local/favorites"),
    "favorites_remove":  ("DELETE","/api/local/favorites/{fav_id}"),
    "notifications_query": ("GET","/api/local/notifications"),

    # MCP 功能 → 调 MCP 客户端
    "do_checkin":           ("MCP", "do_checkin"),
    "get_schedule":         ("MCP", "get_schedule"),
    "get_pending_homeworks":("MCP", "get_pending_homeworks"),
    # ...
}

async def execute_tool(tool_name: str, args: dict) -> str:
    route = TOOL_ROUTING.get(tool_name)
    if not route:
        return f"❌ 未知工具：{tool_name}"

    mode, target = route

    if mode == "MCP":
        # 走 MCP 协议调校园端
        return await mcp_client.call_tool(target, args)

    if mode in ("GET", "POST", "DELETE"):
        # 走本地 REST API（内部 HTTP 或直接调 Service）
        url = _build_url(target, args)
        if mode == "GET":
            async with httpx.AsyncClient() as client:
                resp = await client.get(f"http://localhost:8002{url}", params=args)
                return resp.text
        elif mode == "POST":
            async with httpx.AsyncClient() as client:
                resp = await client.post(f"http://localhost:8002{url}", json=args)
                return resp.text
        else:  # DELETE
            async with httpx.AsyncClient() as client:
                resp = await client.delete(f"http://localhost:8002{url}")
                return resp.text

    return f"❌ 路由错误：{tool_name}"
```

## 🔗 自动操作记录——MCP 和本地统一

每次 Agent 执行完操作（无论是 MCP 还是本地），后端自动写入 `operation_logs`：

```python
# backend/router/chat.py — 工具执行后自动记录
async def execute_and_log(tool_name: str, args: dict, student_id: str, db: Session):
    result = await execute_tool(tool_name, args)

    # 自动写日志
    log_entry = {
        "action_type": _categorize(tool_name),
        "content": _describe(tool_name, args, result),
        "result": result[:200] if result else "",
        "source": "mcp" if tool_name in MCP_TOOLS_NAMES else "local",
    }
    db.add(OperationLog(**log_entry))
    db.commit()

    return result

def _categorize(tool_name: str) -> str:
    CATEGORIES = {
        "do_checkin": "checkin",
        "get_schedule": "schedule",
        "get_pending_homeworks": "homework",
        "submit_homework": "homework",
        "apply_leave": "leave",
        "memos_create": "memo",
        "memos_delete": "memo",
        "favorites_add": "favorite",
        "favorites_remove": "favorite",
    }
    return CATEGORIES.get(tool_name, "other")

def _describe(tool_name: str, args: dict, result: str) -> str:
    DESCRIPTIONS = {
        "do_checkin": lambda a, r: f"签到课程：{_extract_course(r)}",
        "submit_homework": lambda a, r: f"提交作业ID：{a.get('homework_id', '')}",
        "memos_create": lambda a, r: f"创建备忘：{a.get('title', '无标题')}",
        "favorites_add": lambda a, r: f"收藏课程：{a.get('course_name', '')}",
    }
    desc_fn = DESCRIPTIONS.get(tool_name, lambda a, r: f"{tool_name}: {json.dumps(a)}")
    return desc_fn(args, result)
```

```python
# backend/database/models.py — 本地功能模型
from sqlalchemy import Column, Integer, String, Text, DateTime, Boolean
from datetime import datetime
from database.database import Base

class Memo(Base):
    __tablename__ = "memos"

    id = Column(Integer, primary_key=True, autoincrement=True)
    title = Column(String(200), default="")
    content = Column(Text, default="")
    is_important = Column(Integer, default=0)
    tags = Column(String(500), default="")
    created_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)

class OperationLog(Base):
    __tablename__ = "operation_logs"

    id = Column(Integer, primary_key=True, autoincrement=True)
    action_type = Column(String(50), nullable=False)
    content = Column(String(500), nullable=False)
    result = Column(String(500))
    source = Column(String(20), default="local")
    created_at = Column(DateTime, default=datetime.now)

class FavoriteCourse(Base):
    __tablename__ = "favorite_courses"

    id = Column(Integer, primary_key=True, autoincrement=True)
    course_name = Column(String(200), nullable=False)
    teacher = Column(String(100))
    course_id = Column(Integer)
    added_at = Column(DateTime, default=datetime.now)

class ScheduleCache(Base):
    __tablename__ = "schedule_cache"
    course_id = Column(Integer, primary_key=True)
    name = Column(String(200), nullable=False)
    teacher = Column(String(100))
    location = Column(String(200))
    day_of_week = Column(Integer)
    start_time = Column(String(10))
    end_time = Column(String(10))
    cached_at = Column(DateTime, default=datetime.now)

class NotificationHistory(Base):
    __tablename__ = "notification_history"
    id = Column(Integer, primary_key=True, autoincrement=True)
    campus_notif_id = Column(Integer)
    title = Column(String(200))
    content = Column(Text)
    notif_type = Column(String(50))
    is_read = Column(Integer, default=0)
    received_at = Column(DateTime, default=datetime.now)

class NotificationRead(Base):
    __tablename__ = "notification_reads"
    notification_id = Column(Integer, primary_key=True)
    read_at = Column(DateTime, default=datetime.now)
```
